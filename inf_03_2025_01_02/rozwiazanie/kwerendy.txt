FROM zadania
WHERE zadanie LIKE '%mebli%'
ORDER BY data;

    o.nazwisko,
    COUNT(id_zadania) AS 'Liczba zada≈Ñ'
FROM osoby o
JOIN zadania z ON o.id_osoba = z.osoba_id
GROUP BY o.nazwisko;
