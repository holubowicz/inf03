-- 1
SELECT tytul, plik
FROM zdjecia
WHERE polubienia >= 100;


-- 2
SELECT
	z.plik,
    z.tytul,
    z.polubienia,
    a.imie,
    a.nazwisko
FROM zdjecia z
JOIN autorzy a ON z.autorzy_id = a.id
ORDER BY a.nazwisko;


-- 3
SELECT a.imie, COUNT(z.id)
FROM autorzy a
JOIN zdjecia z ON a.id = z.autorzy_id
GROUP BY a.imie;
